SPED Status Reports VBA Deliverable
==================================

Purpose
-------
This file contains a pasteable Excel VBA implementation for the Phase 1 SPED Status Reports workbook.
It is designed to keep confidential data inside Excel while giving staff an app-like workflow through
UserForms layered over hidden backend sheets.

Important Limitations
---------------------
1. Standard VBA code can be copy/pasted directly into Excel modules.
2. UserForm layouts cannot be recreated by plain text alone, so this file includes:
   - the exact form names
   - the required control names
   - the code-behind to paste into each form
3. Gmail and Google Calendar integration are intentionally excluded from Phase 1.
4. Hidden/protected sheets improve privacy, but Excel is not a hardened security platform.

Suggested Paste Order
---------------------
1. Insert standard modules and paste the code blocks in this order:
   - modSpedConstants
   - modSpedSetup
   - modSpedDates
   - modSpedCases
   - modSpedDashboard
   - modSpedSecurity
   - modSpedLauncher
2. Create the four UserForms named:
   - frmLauncher
   - frmCaseIntake
   - frmCaseUpdate
   - frmAdmin
3. Add the controls listed in the "UserForm Control Manifest" section.
4. Paste the code-behind into each UserForm.
5. Paste the ThisWorkbook event code last.
6. Run InstallSpedStatusReports one time.
7. Replace the sample district, campus, evaluator, password, and admin code values before production use.

Option Explicit

Public Sub InstallSpedStatusReports()
    Application.ScreenUpdating = False
    Application.DisplayAlerts = False

    EnsureWorkbookScaffold
    SeedReferenceData
    ProtectWorkbookApp
    RefreshDashboard

    Application.DisplayAlerts = True
    Application.ScreenUpdating = True

    MsgBox "SPED Status Reports has been installed. Review hidden admin tables, update passwords, and replace sample data before production use.", vbInformation
End Sub

Public Sub EnsureWorkbookScaffold()
    Dim ws As Worksheet

    Set ws = EnsureSheet(SHEET_CASES)
    EnsureTable ws, TABLE_CASES, CaseHeaders()
    ws.Visible = xlSheetVeryHidden

    Set ws = EnsureSheet(SHEET_CASE_DOCUMENTS)
    EnsureTable ws, TABLE_CASE_DOCUMENTS, CaseDocumentHeaders()
    ws.Visible = xlSheetVeryHidden

    Set ws = EnsureSheet(SHEET_DISTRICTS)
    EnsureTable ws, TABLE_DISTRICTS, DistrictHeaders()
    ws.Visible = xlSheetVeryHidden

    Set ws = EnsureSheet(SHEET_CAMPUSES)
    EnsureTable ws, TABLE_CAMPUSES, CampusHeaders()
    ws.Visible = xlSheetVeryHidden

    Set ws = EnsureSheet(SHEET_EVALUATORS)
    EnsureTable ws, TABLE_EVALUATORS, EvaluatorHeaders()
    ws.Visible = xlSheetVeryHidden

    Set ws = EnsureSheet(SHEET_CALENDARS)
    EnsureTable ws, TABLE_CALENDARS, CalendarHeaders()
    ws.Visible = xlSheetVeryHidden

    Set ws = EnsureSheet(SHEET_DASHBOARD)
    ws.Visible = xlSheetVisible
    ws.Activate
End Sub

Public Function EnsureSheet(ByVal sheetName As String) As Worksheet
    On Error Resume Next
    Set EnsureSheet = ThisWorkbook.Worksheets(sheetName)
    On Error GoTo 0

    If EnsureSheet Is Nothing Then
        Set EnsureSheet = ThisWorkbook.Worksheets.Add(After:=ThisWorkbook.Worksheets(ThisWorkbook.Worksheets.Count))
        EnsureSheet.Name = sheetName
    End If
End Function

Public Sub EnsureTable(ByVal ws As Worksheet, ByVal tableName As String, ByVal headers As Variant)
    Dim lo As ListObject
    Dim i As Long
    Dim headerRange As Range

    On Error Resume Next
    Set lo = ws.ListObjects(tableName)
    On Error GoTo 0

    If lo Is Nothing Then
        ws.Cells.Clear
        For i = LBound(headers) To UBound(headers)
            ws.Cells(1, i + 1).Value = headers(i)
        Next i
        Set headerRange = ws.Range(ws.Cells(1, 1), ws.Cells(2, UBound(headers) + 1))
        Set lo = ws.ListObjects.Add(xlSrcRange, headerRange, , xlYes)
        lo.Name = tableName
        If Not lo.DataBodyRange Is Nothing Then lo.DataBodyRange.Delete
        ws.Rows(1).Font.Bold = True
        ws.Cells.EntireColumn.AutoFit
    End If
End Sub

Public Function GetTable(ByVal tableName As String) As ListObject
    Dim ws As Worksheet
    For Each ws In ThisWorkbook.Worksheets
        On Error Resume Next
        Set GetTable = ws.ListObjects(tableName)
        On Error GoTo 0
        If Not GetTable Is Nothing Then Exit Function
    Next ws
End Function

Public Function ColumnIndex(ByVal lo As ListObject, ByVal columnName As String) As Long
    Dim i As Long
    For i = 1 To lo.ListColumns.Count
        If StrComp(lo.ListColumns(i).Name, columnName, vbTextCompare) = 0 Then
            ColumnIndex = i
            Exit Function
        End If
    Next i
    Err.Raise vbObjectError + 1000, "ColumnIndex", "Column not found: " & columnName
End Function

Option Explicit

Public Function GetDistrictRule(ByVal districtName As String, ByVal ruleColumn As String, ByVal fallbackValue As Long) As Long
    Dim lo As ListObject
    Dim dataRow As ListRow

    Set lo = GetTable(TABLE_DISTRICTS)

    For Each dataRow In lo.ListRows
        If StrComp(CStr(dataRow.Range.Cells(1, ColumnIndex(lo, "District")).Value), districtName, vbTextCompare) = 0 Then
            If IsNumeric(dataRow.Range.Cells(1, ColumnIndex(lo, ruleColumn)).Value) Then
                GetDistrictRule = CLng(dataRow.Range.Cells(1, ColumnIndex(lo, ruleColumn)).Value)
                Exit Function
            End If
        End If
    Next dataRow

    GetDistrictRule = fallbackValue
End Function

Public Function IsInstructionalDay(ByVal checkDate As Date, ByVal districtName As String) As Boolean
    Dim weekdayNumber As Long
    Dim lo As ListObject
    Dim dataRow As ListRow

    weekdayNumber = VBA.Weekday(checkDate, vbMonday)
    If weekdayNumber > 5 Then Exit Function

    Set lo = GetTable(TABLE_CALENDARS)
    For Each dataRow In lo.ListRows
        If StrComp(CStr(dataRow.Range.Cells(1, ColumnIndex(lo, "District")).Value), districtName, vbTextCompare) = 0 Then
            If IsDate(dataRow.Range.Cells(1, ColumnIndex(lo, "NonInstructionalDate")).Value) Then
                If CLng(CDate(dataRow.Range.Cells(1, ColumnIndex(lo, "NonInstructionalDate")).Value)) = CLng(checkDate) Then
                    Exit Function
                End If
            End If
        End If
    Next dataRow

    IsInstructionalDay = True
End Function

Public Function AddInstructionalDays(ByVal startDate As Date, ByVal daysToAdd As Long, ByVal districtName As String) As Date
    Dim cursor As Date
    Dim counted As Long

    cursor = startDate
    counted = 0

    Do While counted < daysToAdd
        cursor = cursor + 1
        If IsInstructionalDay(cursor, districtName) Then counted = counted + 1
    Loop

    AddInstructionalDays = cursor
End Function

Public Function BuildProjectedDates(ByVal caseType As String, _
                                    ByVal districtName As String, _
                                    ByVal referralDate As Variant, _
                                    ByVal actualConsentDate As Variant, _
                                    ByVal actualFIIEDate As Variant, _
                                    ByVal reevalDueDate As Variant) As Object
    Dim result As Object
    Dim responseDue As Variant
    Dim consentAnchor As Variant
    Dim fiieAnchor As Variant
    Dim fiieDue As Variant
    Dim ardDue As Variant

    Set result = CreateObject("Scripting.Dictionary")
    result("ResponseDueDate") = vbNullString
    result("ProjectedConsentDate") = vbNullString
    result("ProjectedFIIEDueDate") = vbNullString
    result("ProjectedARDDueDate") = vbNullString

    If StrComp(caseType, CASE_TYPE_INITIAL, vbTextCompare) = 0 Then
        If IsDate(referralDate) Then
            responseDue = AddInstructionalDays(CDate(referralDate), GetDistrictRule(districtName, "ResponseSchoolDays", 15), districtName)
            result("ResponseDueDate") = responseDue
            result("ProjectedConsentDate") = responseDue

            If IsDate(actualConsentDate) Then
                consentAnchor = CDate(actualConsentDate)
            Else
                consentAnchor = responseDue
            End If

            fiieDue = AddInstructionalDays(CDate(consentAnchor), GetDistrictRule(districtName, "FIIESchoolDays", 45), districtName)
            result("ProjectedFIIEDueDate") = fiieDue

            If IsDate(actualFIIEDate) Then
                fiieAnchor = CDate(actualFIIEDate)
            Else
                fiieAnchor = fiieDue
            End If

            ardDue = DateAdd("d", GetDistrictRule(districtName, "ARDCalendarDays", 30), CDate(fiieAnchor))
            result("ProjectedARDDueDate") = ardDue
        End If
    Else
        If IsDate(reevalDueDate) Then
            result("ProjectedFIIEDueDate") = CDate(reevalDueDate)
            If IsDate(actualFIIEDate) Then
                result("ProjectedARDDueDate") = DateAdd("d", GetDistrictRule(districtName, "ARDCalendarDays", 30), CDate(actualFIIEDate))
            Else
                result("ProjectedARDDueDate") = DateAdd("d", GetDistrictRule(districtName, "ARDCalendarDays", 30), CDate(reevalDueDate))
            End If
        End If
    End If

    Set BuildProjectedDates = result
End Function

Public Function FormatDateOrBlank(ByVal inputValue As Variant) As String
    If IsDate(inputValue) Then
        FormatDateOrBlank = Format$(CDate(inputValue), "mm/dd/yyyy")
    Else
        FormatDateOrBlank = vbNullString
    End If
End Function

Option Explicit

Public Function GenerateCaseID(ByVal caseType As String) As String
    Dim prefix As String
    prefix = "INI"
    If StrComp(caseType, CASE_TYPE_REEVALUATION, vbTextCompare) = 0 Then prefix = "REE"
    GenerateCaseID = prefix & "-" & Format$(Now, "yyyymmdd-hhnnss")
End Function

Public Sub PopulateReferenceLists(ByVal frm As Object)
    PopulateComboFromTable frm.cboCampus, TABLE_CAMPUSES, "Campus", "Active", "Yes"
    PopulateComboFromTable frm.cboDistrict, TABLE_DISTRICTS, "District", "Active", "Yes"

    If Len(Trim$(SafeControlValue(frm, "cboDistrict"))) > 0 Then
        PopulateEvaluatorsForDistrict frm, SafeControlValue(frm, "cboDistrict")
    Else
        PopulateComboFromTable frm.cboLeadEvaluator, TABLE_EVALUATORS, "LeadEvaluator", "Active", "Yes"
    End If
End Sub

Public Sub PopulateEvaluatorsForDistrict(ByVal frm As Object, ByVal districtName As String)
    Dim lo As ListObject
    Dim dataRow As ListRow

    frm.cboLeadEvaluator.Clear
    Set lo = GetTable(TABLE_EVALUATORS)

    For Each dataRow In lo.ListRows
        If StrComp(CStr(dataRow.Range.Cells(1, ColumnIndex(lo, "Active")).Value), "Yes", vbTextCompare) = 0 Then
            If StrComp(CStr(dataRow.Range.Cells(1, ColumnIndex(lo, "District")).Value), districtName, vbTextCompare) = 0 Then
                frm.cboLeadEvaluator.AddItem CStr(dataRow.Range.Cells(1, ColumnIndex(lo, "LeadEvaluator")).Value)
            End If
        End If
    Next dataRow
End Sub

Public Sub PopulateComboFromTable(ByVal cbo As Object, ByVal tableName As String, ByVal valueColumn As String, _
                                  Optional ByVal filterColumn As String = vbNullString, _
                                  Optional ByVal filterValue As String = vbNullString)
    Dim lo As ListObject
    Dim dataRow As ListRow
    Dim includeRow As Boolean

    cbo.Clear
    Set lo = GetTable(tableName)

    For Each dataRow In lo.ListRows
        includeRow = True
        If Len(filterColumn) > 0 Then
            includeRow = StrComp(CStr(dataRow.Range.Cells(1, ColumnIndex(lo, filterColumn)).Value), filterValue, vbTextCompare) = 0
        End If
        If includeRow Then cbo.AddItem CStr(dataRow.Range.Cells(1, ColumnIndex(lo, valueColumn)).Value)
    Next dataRow
End Sub

Public Function SafeControlValue(ByVal frm As Object, ByVal controlName As String) As String
    On Error Resume Next
    SafeControlValue = Trim$(CStr(frm.Controls(controlName).Value))
    On Error GoTo 0
End Function

Public Function ResolveDistrictFromCampus(ByVal campusName As String) As String

Public Function DuplicateOpenCaseExists(ByVal studentID As String, ByVal caseType As String, Optional ByVal ignoreCaseID As String = vbNullString) As Boolean
    Dim lo As ListObject
    Dim dataRow As ListRow

    Set lo = GetTable(TABLE_CASES)

    For Each dataRow In lo.ListRows
        If StrComp(CStr(dataRow.Range.Cells(1, ColumnIndex(lo, "StudentID")).Value), studentID, vbTextCompare) = 0 Then
            If StrComp(CStr(dataRow.Range.Cells(1, ColumnIndex(lo, "CaseType")).Value), caseType, vbTextCompare) = 0 Then
                If StrComp(CStr(dataRow.Range.Cells(1, ColumnIndex(lo, "Status")).Value), STATUS_COMPLETE, vbTextCompare) <> 0 Then
                    If StrComp(CStr(dataRow.Range.Cells(1, ColumnIndex(lo, "CaseID")).Value), ignoreCaseID, vbTextCompare) <> 0 Then
                        DuplicateOpenCaseExists = True
                        Exit Function
                    End If
                End If
            End If
        End If
    Next dataRow
End Function

Public Function DetermineStatus(ByVal actualConsentDate As Variant, ByVal actualARDDate As Variant) As String
    If IsDate(actualARDDate) Then
        DetermineStatus = STATUS_COMPLETE
    ElseIf IsDate(actualConsentDate) Then
        DetermineStatus = STATUS_IN_PROGRESS
    Else
        DetermineStatus = STATUS_REFERRAL_RECEIVED
    End If
End Function

Public Function DeterminePrimaryDeadline(ByVal caseType As String, ByVal responseDueDate As Variant, _
                                         ByVal projectedFIIEDueDate As Variant, ByVal projectedARDDueDate As Variant, _
                                         ByVal actualConsentDate As Variant, ByVal actualFIIEDate As Variant, _
                                         ByVal actualARDDate As Variant, ByVal reevalDueDate As Variant, _
                                         ByVal manualPrimaryDeadline As Variant) As Variant
    If IsDate(manualPrimaryDeadline) Then
        DeterminePrimaryDeadline = CDate(manualPrimaryDeadline)
        Exit Function
    End If

    If StrComp(caseType, CASE_TYPE_INITIAL, vbTextCompare) = 0 Then
        If Not IsDate(actualConsentDate) Then
            DeterminePrimaryDeadline = responseDueDate
        ElseIf Not IsDate(actualFIIEDate) Then
            DeterminePrimaryDeadline = projectedFIIEDueDate
        ElseIf Not IsDate(actualARDDate) Then
            DeterminePrimaryDeadline = projectedARDDueDate
        Else
            DeterminePrimaryDeadline = actualARDDate
        End If
    Else
        If Not IsDate(actualFIIEDate) Then
            DeterminePrimaryDeadline = reevalDueDate
        ElseIf Not IsDate(actualARDDate) Then
            DeterminePrimaryDeadline = projectedARDDueDate
        Else
            DeterminePrimaryDeadline = actualARDDate
        End If
    End If
End Function

Public Function SaveCaseFromIntakeForm(ByVal frm As Object) As Boolean
    Dim lo As ListObject
    Dim rowRange As Range
    Dim caseType As String
    Dim districtName As String
    Dim caseID As String
    Dim projectedDates As Object
    Dim referralDate As Variant
    Dim reevalDueDate As Variant
    Dim actualConsentDate As Variant
    Dim actualFIIEDate As Variant
    Dim actualARDDate As Variant

    caseType = frm.Tag
    districtName = Trim$(frm.cboDistrict.Value)

    If Len(Trim$(frm.txtStudentName.Value)) = 0 Then
        MsgBox "Student Name is required.", vbExclamation
        Exit Function
    End If

    If Len(Trim$(frm.txtStudentID.Value)) = 0 Then
        MsgBox "Student ID is required.", vbExclamation
        Exit Function
    End If

    If Len(districtName) = 0 Then
        MsgBox "District is required.", vbExclamation
        Exit Function
    End If

    If DuplicateOpenCaseExists(frm.txtStudentID.Value, caseType) Then
        MsgBox "An open " & caseType & " case already exists for this student.", vbExclamation
        Exit Function
    End If

    referralDate = ParseDateOrBlank(frm.txtReferralReceivedDate.Value)
    reevalDueDate = ParseDateOrBlank(frm.txtReevalDueDate.Value)
    actualConsentDate = ParseDateOrBlank(frm.txtActualConsentDate.Value)
    actualFIIEDate = ParseDateOrBlank(frm.txtActualFIIEDate.Value)
    actualARDDate = ParseDateOrBlank(frm.txtActualARDDate.Value)

    If StrComp(caseType, CASE_TYPE_INITIAL, vbTextCompare) = 0 And Not IsDate(referralDate) Then
        MsgBox "Referral Received Date is required for Initial cases.", vbExclamation
        Exit Function
    End If

    If StrComp(caseType, CASE_TYPE_REEVALUATION, vbTextCompare) = 0 And Not IsDate(reevalDueDate) Then
        MsgBox "Re-evaluation Due Date is required for Re-evaluation cases.", vbExclamation
        Exit Function
    End If

    Set projectedDates = BuildProjectedDates(caseType, districtName, referralDate, actualConsentDate, actualFIIEDate, reevalDueDate)

    Set lo = GetTable(TABLE_CASES)
    Set rowRange = lo.ListRows.Add.Range

    caseID = GenerateCaseID(caseType)

    rowRange.Cells(1, ColumnIndex(lo, "CaseID")).Value = caseID
    rowRange.Cells(1, ColumnIndex(lo, "CaseType")).Value = caseType
    rowRange.Cells(1, ColumnIndex(lo, "StudentName")).Value = Trim$(frm.txtStudentName.Value)

Public Sub ReplaceCaseDocuments(ByVal caseID As String, ByVal rawDocumentText As String)
    Dim lo As ListObject
    Dim i As Long
    Dim documentLines() As String
    Dim lineText As String
    Dim parts() As String
    Dim rowRange As Range
    Dim labelText As String
    Dim pathText As String
    Dim docCounter As Long

    Set lo = GetTable(TABLE_CASE_DOCUMENTS)

    For i = lo.ListRows.Count To 1 Step -1
        If StrComp(CStr(lo.ListRows(i).Range.Cells(1, ColumnIndex(lo, "CaseID")).Value), caseID, vbTextCompare) = 0 Then
            lo.ListRows(i).Delete
        End If
    Next i

    rawDocumentText = Replace(rawDocumentText, vbCrLf, vbLf)
    rawDocumentText = Replace(rawDocumentText, vbCr, vbLf)
    documentLines = Split(rawDocumentText, vbLf)
    docCounter = 0

    For i = LBound(documentLines) To UBound(documentLines)
        lineText = Trim$(documentLines(i))
        If Len(lineText) > 0 Then
            docCounter = docCounter + 1
            parts = Split(lineText, "|")
            If UBound(parts) >= 1 Then
                labelText = Trim$(parts(0))
                pathText = Trim$(Mid$(lineText, InStr(1, lineText, "|") + 1))
            Else
                labelText = "Document " & docCounter
                pathText = lineText
            End If

            Set rowRange = lo.ListRows.Add.Range
            rowRange.Cells(1, ColumnIndex(lo, "DocumentID")).Value = caseID & "-DOC" & Format$(docCounter, "00")
            rowRange.Cells(1, ColumnIndex(lo, "CaseID")).Value = caseID
            rowRange.Cells(1, ColumnIndex(lo, "DocumentLabel")).Value = labelText
            rowRange.Cells(1, ColumnIndex(lo, "DocumentPath")).Value = pathText
            rowRange.Cells(1, ColumnIndex(lo, "AddedAt")).Value = Now
        End If
    Next i
End Sub

Public Function GetCaseDocumentText(ByVal caseID As String) As String
    Dim lo As ListObject
    Dim dataRow As ListRow
    Dim lines As String

    Set lo = GetTable(TABLE_CASE_DOCUMENTS)

    For Each dataRow In lo.ListRows
        If StrComp(CStr(dataRow.Range.Cells(1, ColumnIndex(lo, "CaseID")).Value), caseID, vbTextCompare) = 0 Then
            lines = lines & CStr(dataRow.Range.Cells(1, ColumnIndex(lo, "DocumentLabel")).Value) & "|" & _
                    CStr(dataRow.Range.Cells(1, ColumnIndex(lo, "DocumentPath")).Value) & vbCrLf
        End If
    Next dataRow

    If Len(lines) > 0 Then lines = Left$(lines, Len(lines) - 2)
    GetCaseDocumentText = lines
End Function

        ws.Cells(rowNumber, 8).Value = statusValue
        ws.Cells(rowNumber, 9).Value = nextDeadline
        ws.Cells(rowNumber, 10).Value = dataRow.Range.Cells(1, ColumnIndex(lo, "ResponseDueDate")).Value
        ws.Cells(rowNumber, 11).Value = dataRow.Range.Cells(1, ColumnIndex(lo, "ProjectedFIIEDueDate")).Value
        ws.Cells(rowNumber, 12).Value = dataRow.Range.Cells(1, ColumnIndex(lo, "ProjectedARDDueDate")).Value

        ws.Cells(rowNumber, 14).Value = dataRow.Range.Cells(1, ColumnIndex(lo, "UpdatedAt")).Value

        ApplyDashboardRowColor ws.Rows(rowNumber), statusValue, nextDeadline
        rowNumber = rowNumber + 1
    Next dataRow

        ws.Range("A4:N" & (rowNumber - 1)).AutoFilter

    End If

Private Sub ApplyDashboardRowColor(ByVal targetRow As Range, ByVal statusValue As String, ByVal deadlineValue As Variant)
    targetRow.Interior.Pattern = xlSolid

    If StrComp(statusValue, STATUS_COMPLETE, vbTextCompare) = 0 Then
        targetRow.Interior.Color = RGB(198, 239, 206)
    ElseIf IsDate(deadlineValue) Then
        If CDate(deadlineValue) < Date Then
            targetRow.Interior.Color = RGB(255, 199, 206)
        ElseIf DateDiff("d", Date, CDate(deadlineValue)) <= 7 Then
            targetRow.Interior.Color = RGB(255, 235, 156)
        Else
            targetRow.Interior.ColorIndex = xlColorIndexNone
        End If
    Else
        targetRow.Interior.ColorIndex = xlColorIndexNone
    End If
End Sub

Recommended Field Behavior
--------------------------
- Initial case
  - Enter Referral Received Date at intake.
  - Response Due Date auto-calculates to 15 instructional days using the district calendar.
  - Projected Consent Date mirrors Response Due Date until actual consent is entered.
  - Projected FIIE Due Date recalculates from the consent anchor.
  - Projected ARD Due Date recalculates from the FIIE completion anchor.
- Re-evaluation case
  - Enter the Re-evaluation Due Date at intake.
  - The dashboard uses that due date as the primary deadline until the FIIE milestone is entered.
- Milestone updates
  - `Referral Received` until Actual Consent Date exists
  - `In Progress` once Actual Consent Date exists
  - `Complete` once Actual ARD Date exists
- Documents
  - Store one or more file paths or network links as text references
  - No files are embedded in the workbook
